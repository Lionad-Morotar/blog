get-intrinsic

get-intrinsic жҳҜд»Җд№Ҳпјҹ

get-intrinsic жҳҜ ljharb ејҖеҸ‘зҡ„дёҖдёӘ JavaScript е·Ҙе…·еә“пјҢз”ЁдәҺеңЁйҰ–ж¬Ў require ж—¶иҺ·еҸ–е№¶зЁіеҒҘең°зј“еӯҳжүҖжңү JavaScript иҜӯиЁҖзә§еҶ…зҪ®ж–№жі•пјҲintrinsicsпјүгҖӮе®ғзҡ„ж ёеҝғз”ЁйҖ”жҳҜеңЁиҝҗиЎҢж—¶е®үе…Ёең°иҺ·еҸ– JavaScript еҶ…зҪ®еҜ№иұЎе’Ңж–№жі•зҡ„еј•з”ЁпјҢеҚідҪҝиҝҷдәӣж–№жі•еңЁиҝҗиЎҢж—¶иў«зҜЎж”№жҲ–еҲ йҷӨгҖӮ

и§Ғпјҡget-intrinsic

дёәд»Җд№ҲйңҖиҰҒ get-intrinsicпјҹ

еңЁ JavaScript дёӯпјҢе…ЁеұҖеҜ№иұЎе’Ңе…¶ж–№жі•еҸҜиғҪиў«жҒ¶ж„Ҹд»Јз ҒзҜЎж”№пјҲеҰӮеҺҹеһӢжұЎжҹ“ж”»еҮ»пјүгҖӮget-intrinsic жҸҗдҫӣдәҶдёҖз§ҚйҳІеҫЎжҖ§зј–зЁӢж–№ејҸпјҡ

  • зј“еӯҳжңәеҲ¶пјҡеңЁйҰ–ж¬ЎеҠ иҪҪж—¶зј“еӯҳеҶ…зҪ®ж–№жі•зҡ„еҺҹе§Ӣеј•з”Ё
  • йҳІзҜЎж”№пјҡеҚідҪҝе…ЁеұҖеҜ№иұЎзҡ„ж–№жі•иў«еҲ йҷӨжҲ–дҝ®ж”№пјҢд»ҚиғҪйҖҡиҝҮзј“еӯҳдҪҝз”ЁеҺҹе§ӢеҠҹиғҪ
  • и§„иҢғж”ҜжҢҒпјҡж”ҜжҢҒ JavaScript и§„иҢғдёӯе®ҡд№үзҡ„жүҖжңү intrinsics иҜӯжі•пјҲеҰӮ %Array.prototype.push%пјү

еҰӮдҪ•дҪҝз”Ё get-intrinsicпјҹ

var GetIntrinsic = require('get-intrinsic');

// иҺ·еҸ–йқҷжҖҒж–№жі•
GetIntrinsic('%Math.pow%'); // Math.pow зҡ„зј“еӯҳеј•з”Ё

// иҺ·еҸ–е®һдҫӢж–№жі•
GetIntrinsic('%Array.prototype.push%'); // Array.prototype.push зҡ„зј“еӯҳеј•з”Ё

// еҚідҪҝеҺҹж–№жі•иў«еҲ йҷӨпјҢзј“еӯҳзҡ„ж–№жі•д»ҚеҸҜе·ҘдҪң
delete Array.prototype.push;
GetIntrinsic('%Array.prototype.push%').call(arr, 4); // д»Қз„¶жңүж•Ҳ

еҗҢзұ»е·Ҙе…·еҜ№жҜ”

е·Ҙе…·/ж–№жЎҲзұ»еһӢеҠҹиғҪзү№зӮ№дҪҝз”ЁеңәжҷҜ
get-intrinsicnpm еә“зј“еӯҳе№¶жҡҙйңІжүҖжңү ES IntrinsicsпјҢж”ҜжҢҒ %Array.prototype.push% иҜӯжі•йңҖиҰҒе®үе…ЁдҪҝз”ЁеҶ…зҪ®ж–№жі•зҡ„еңәжҷҜ
es5-shim / es6-shimpolyfillдёәж—§зҺҜеўғжҸҗдҫӣж–° API е®һзҺ°пјҢеҶ…йғЁзј“еӯҳеҺҹе§Ӣж–№жі•еј•з”Ёе…је®№ж—§жөҸи§ҲеҷЁзҡ„ polyfill ж–№жЎҲ
core-jspolyfill еә“жҸҗдҫӣж ҮеҮҶеә“ polyfillпјҢеҸҜжЈҖжөӢеҺҹз”ҹж–№жі•жҳҜеҗҰиў«зҜЎж”№пјҲ.sham еұһжҖ§пјүBabel иҪ¬иҜ‘еҗҺзҡ„ polyfill зҺҜеўғ

Object.freezeеҺҹз”ҹж–№жі•еҶ»з»“е…ЁеұҖеҜ№иұЎеҺҹеһӢпјҢйҳІжӯўиў«зҜЎж”№з®ҖеҚ•йҳІжҠӨпјҲдҪҶдјҡз ҙеқҸеҰӮ Vue2 зӯүдҫқиө–еҺҹеһӢдҝ®ж”№зҡ„жЎҶжһ¶пјү

  • дё“й—Ёи®ҫи®Ўз”ЁдәҺ иҺ·еҸ–е’Ңзј“еӯҳ intrinsicsпјҢиҖҢйқһжЈҖжөӢжҲ– polyfill

get-intrinsic зҡ„ж ёеҝғжҳҜдёҖдёӘе·ЁеӨ§зҡ„ INTRINSICS еҜ№иұЎпјҢеңЁжЁЎеқ—еҠ иҪҪж—¶зј“еӯҳжүҖжңүеҶ…зҪ®ж–№жі•пјҡ

var $Error = require('es-errors');             // Error жһ„йҖ еҮҪж•°

module.exports = function GetIntrinsic(name, allowMissing) {
  var parts = stringToPath(name);
  var value = INTRINSICS['%' + parts[0] + '%'];

3. Object еҺҹеӯҗж“ҚдҪңпјҲes-object-atomsпјү

е®һзҺ° ES и§„иҢғдёӯзҡ„ Object зӣёе…іжҠҪиұЎж“ҚдҪңпјҡ

// es-object-atoms/index.js
'use strict';

module.exports = Object;

// es-object-atoms/ToObject.js
'use strict';
var $Object = require('./');

// ES и§„иҢғ: ToObject жҠҪиұЎж“ҚдҪң
module.exports = function ToObject(value) {
  if (value == null) {  // null жҲ– undefined
    throw new TypeError('Cannot convert undefined or null to object');
  }
  return $Object(value);
};

// es-object-atoms/RequireObjectCoercible.js
'use strict';

// ES и§„иҢғ: RequireObjectCoercible жҠҪиұЎж“ҚдҪң
module.exports = function RequireObjectCoercible(value) {
  if (value == null) {
    throw new TypeError('Cannot call method on ' + value);
  }
  return value;
};

// es-object-atoms/isObject.js
'use strict';

module.exports = function isObject(value) {
  return typeof value === 'object' && value !== null;
};

и§„иҢғжҳ е°„пјҡ

7. и·Ҝеҫ„и§ЈжһҗдёҺеҲ«еҗҚжҳ е°„

get-intrinsic ж”ҜжҢҒдёӨз§ҚиҜӯжі•пјҡ%Array.prototype.push% е’Ң %ArrayProto_push%пјҡ

// еҲ«еҗҚжҳ е°„иЎЁ
var LEGACY_ALIASES = {
  '%ArrayPrototype%': ['Array', 'prototype'],
  '%ArrayProto_push%': ['Array', 'prototype', 'push'],
  '%ObjProto_toString%': ['Object', 'prototype', 'toString'],
};

// и·Ҝеҫ„еӯ—з¬ҰдёІи§ЈжһҗпјҲеӨ„зҗҶеј•еҸ·е’ҢиҪ¬д№үпјү
var rePropName = /[^.[\]]+|\[(?:(-?\d+(?:\.\d+)?)|(["'])((?:(?!\2)[^\\]|\\.)*?)\2)\]|(?=(?:\.|\[\])(?:\.|\[\]|%$))/g;

function stringToPath(string) {
  var result = [];
  string.replace(rePropName, function(match, number, quote, subString) {
    result.push(quote ? subString.replace(/\\(\)?/g, '$1') : (number || match));
  });
  return result;
}
